class Solution {
    public boolean canPartition(int[] nums) {
        int len = nums.length;
        int sum =0;
        for(Integer num:nums){
            sum += num;
        }
        if(sum%2!=0) return false;
        int half = sum/2;
        return dfs(nums,0,0,half);
    }
    
    public boolean dfs(int[] nums,int cur,int sum,int half){
        boolean res = false;
        if(cur!=nums.length && sum==half) return true;
        else if(cur>nums.length-1) return false;
        res |= dfs(nums, cur+1,sum+nums[cur],half);
        if(res) return res;
        res |= dfs(nums,cur+1,sum,half);
        return res;
    }
}
